## Install

```bash
yarn add react-to-image
```

or

```bash
npm --save react-to-image
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-to-image'
import 'react-to-image/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
